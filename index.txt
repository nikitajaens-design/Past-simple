    <script>
        const verbsData = [
            { id: 1, base: "Write", past: "wrote", pastOptions: ["wrote"] },
            { id: 2, base: "Look", past: "looked", pastOptions: ["looked"] },
            { id: 3, base: "Wake up", past: "woke up", pastOptions: ["woke up", "waked up"] },
            { id: 4, base: "Want", past: "wanted", pastOptions: ["wanted"] },
            { id: 5, base: "Read", past: "read", pastOptions: ["read"] },
            { id: 6, base: "Need", past: "needed", pastOptions: ["needed"] },
            { id: 7, base: "Take", past: "took", pastOptions: ["took"] },
            { id: 8, base: "Eat", past: "ate", pastOptions: ["ate"] },
            { id: 9, base: "Live", past: "lived", pastOptions: ["lived"] },
            { id: 10, base: "Cook", past: "cooked", pastOptions: ["cooked"] }
        ];

        let completedVerbs = 0;
        let completedSentences = 0;
        const checkedItems = new Set();

        function initActivity() {
            const verbsContainer = document.getElementById('verbs-container');
            const sentencesContainer = document.getElementById('sentences-container');
            
            verbsContainer.innerHTML = '';
            sentencesContainer.innerHTML = '';
            completedVerbs = 0;
            completedSentences = 0;
            checkedItems.clear();

            verbsData.forEach((verb, index) => {
                // Verb conjugation item
                const verbDiv = document.createElement('div');
                verbDiv.className = 'verb-item';
                verbDiv.innerHTML = `
                    <div class="number">${index + 1}</div>
                    <div class="base-verb">${verb.base}</div>
                    <input type="text" 
                           class="past-input" 
                           id="verb-${verb.id}" 
                           placeholder="Past tense..."
                           onkeypress="if(event.key === 'Enter') checkVerb(${verb.id})">
                    <button class="check-btn" onclick="checkVerb(${verb.id})">Check</button>
                    <span class="status" id="status-verb-${verb.id}"></span>
                    <div class="feedback" id="feedback-verb-${verb.id}"></div>
                `;
                verbsContainer.appendChild(verbDiv);

                // Sentence writing item
                const sentenceDiv = document.createElement('div');
                sentenceDiv.className = 'sentence-item';
                sentenceDiv.innerHTML = `
                    <div class="sentence-label">
                        <span>üéØ</span>
                        <span>Write a sentence with <strong>${verb.base}</strong> (past tense)</span>
                    </div>
                    <textarea class="sentence-input" 
                              id="sentence-${verb.id}" 
                              placeholder="Example: Yesterday I ${verb.past}..."
                              rows="2"></textarea>
                    <div class="hint">üí° Hint: Include a time expression like yesterday, last week, ago...</div>
                    <div class="action-buttons">
                        <button class="btn-check-sentence" onclick="checkSentence(${verb.id})">Check Sentence</button>
                        <span class="status" id="status-sentence-${verb.id}" style="font-size: 20px;"></span>
                    </div>
                    <div class="feedback" id="feedback-sentence-${verb.id}"></div>
                `;
                sentencesContainer.appendChild(sentenceDiv);
            });

            updateProgress();
        }

        function checkVerb(id) {
            const input = document.getElementById(`verb-${id}`);
            const status = document.getElementById(`status-verb-${id}`);
            const feedback = document.getElementById(`feedback-verb-${id}`);
            const userAnswer = input.value.trim().toLowerCase();
            const verb = verbsData.find(v => v.id === id);

            if (!userAnswer) {
                feedback.textContent = 'Please write an answer';
                feedback.className = 'feedback error';
                return;
            }

            const isCorrect = verb.pastOptions.includes(userAnswer) || 
                             verb.pastOptions.some(opt => userAnswer === opt.toLowerCase());

            if (isCorrect) {
                status.textContent = '‚úÖ';
                input.className = 'past-input correct';
                feedback.textContent = `Correct! "${verb.past}"`;
                feedback.className = 'feedback success';
                
                if (!checkedItems.has(`verb-${id}`)) {
                    checkedItems.add(`verb-${id}`);
                    completedVerbs++;
                    updateProgress();
                }
            } else {
                status.textContent = '‚ùå';
                input.className = 'past-input wrong';
                feedback.textContent = 'Not quite. Try again!';
                feedback.className = 'feedback error';
                setTimeout(() => {
                    input.className = 'past-input';
                }, 1000);
            }

            checkCompletion();
        }

        function checkSentence(id) {
            const input = document.getElementById(`sentence-${id}`);
            const status = document.getElementById(`status-sentence-${id}`);
            const feedback = document.getElementById(`feedback-sentence-${id}`);
            const userSentence = input.value.trim();
            const verb = verbsData.find(v => v.id === id);

            if (!userSentence || userSentence.length < 5) {
                feedback.textContent = 'Please write a complete sentence (at least 5 words)';
                feedback.className = 'feedback error';
                return;
            }

            // Check if sentence contains the past tense verb
            const pastTense = verb.past.toLowerCase();
            const hasVerb = userSentence.toLowerCase().includes(pastTense);
            
            // Check for time expressions
            const timeExpressions = ['yesterday', 'last', 'ago', 'this morning', 'last night', 'last week', 'last month', 'last year', 'earlier', 'before'];
            const hasTimeExpression = timeExpressions.some(expr => userSentence.toLowerCase().includes(expr));

            // Basic validation: starts with capital letter and ends with period
            const startsWithCapital = /^[A-Z]/.test(userSentence);
            const endsWithPunctuation = /[.!?]$/.test(userSentence);

            if (hasVerb && startsWithCapital && endsWithPunctuation) {
                status.textContent = '‚úÖ';
                input.className = 'sentence-input correct';
                
                let msg = 'Excellent sentence!';
                if (hasTimeExpression) msg += ' Great use of time expression!';
                feedback.textContent = msg;
                feedback.className = 'feedback success';
                
                if (!checkedItems.has(`sentence-${id}`)) {
                    checkedItems.add(`sentence-${id}`);
                    completedSentences++;
                    updateProgress();
                }
            } else {
                status.textContent = '‚ùå';
                input.className = 'sentence-input wrong';
                
                let errors = [];
                if (!hasVerb) errors.push(`use "${verb.past}"`);
                if (!startsWithCapital) errors.push('start with capital letter');
                if (!endsWithPunctuation) errors.push('end with punctuation');
                
                feedback.textContent = `Please: ${errors.join(', ')}`;
                feedback.className = 'feedback error';
                
                setTimeout(() => {
                    input.className = 'sentence-input';
                }, 1500);
            }

            checkCompletion();
        }

        function updateProgress() {
            const total = verbsData.length * 2;
            const completed = completedVerbs + completedSentences;
            const percentage = (completed / total) * 100;
            
            document.getElementById('progress-bar').style.width = `${percentage}%`;
            document.getElementById('progress-text').textContent = `${completed}/${total} completed`;
        }

        function checkCompletion() {
            const total = verbsData.length * 2;
            const completed = completedVerbs + completedSentences;
            
            if (completed === total) {
                setTimeout(() => {
                    document.getElementById('results').style.display = 'block';
                    document.getElementById('final-score').textContent = `${completed}/${total}`;
                    
                    const messages = [
                        "Great job! You've mastered the simple past! üåü",
                        "Excellent work! Your English is improving! üèÜ",
                        "Fantastic! Keep practicing! üí™",
                        "Perfect! You're doing amazing! ‚ú®"
                    ];
                    document.getElementById('result-message').textContent = messages[Math.floor(Math.random() * messages.length)];
                    document.getElementById('results').scrollIntoView({ behavior: 'smooth' });
                }, 500);
            }
        }

        function resetActivity() {
            document.getElementById('results').style.display = 'none';
            initActivity();
        }

        window.onload = initActivity;
    </script>
